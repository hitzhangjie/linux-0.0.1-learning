 77 /**
 78  * hit.zhangjie@gmail.com 
 79  * 2015-04-10 06:21:14 PM
 80  * 
 81  * in 80386, to implement 'protective memory access mode', intel adds some
 82  * other registers to achieve this.
 83  * 1) GDTR, short for global descriptor table register, which store a base
 84  * address which points to an array in which every element is a special struct.
 85  * this special struct contains segment details, including value of
 86  * cs or ds, the length of this segment, and the privelege to access this
 87  * segment.
 88  * 2) LDTR, short for local descriptor table register, which function
 89  * similarly to the GDTR. GDTR is shared between kernel and all processes,
 90  * while LDTR belongs to one specific process.
 91  * also, in linux, processes don't use LDTR, it's only used in vm86 model, for
 92  * example, when we use WINE to run some windows applications, LDTR will be
 93  * used.
 94  * 3) CS/DS..., now in protective model, value of CS/DS... is not the actual
 95  * value of code segment/data segment..., the value in them is just the index
 96  * of the array pointed by the value of GDTR.
 97  * for example, the struct pointed by (GDTR)[(CS)] is the actual segment
 98  * details, from this struct, we can get segment's base address and length.
 99  * combined with IP, we can take instructions from the code segment. 
100  * also, notice that, LGDT\SGDT are privilege instructions, any other
101  * instruction can't access/modify their value, because of this basic
102  * and other mechanisms, protective memory access model is achieved.
103  *
104  * in GDT, here, GDT is the array pointed by (GDTR). every element in it is
105  * either an LDTR, or a task struct. here, terminology 'task' refers to
106  * 'process'.
107  * every process has 2 structs in the GDT, one is tss struct(tss, short for
108  * Task State Segment), one is LDTR whose value points LDT owned by process.
109  * GDT[0]\[1] is cleared to zero, it's useless.
110  * GDT[2]\[3] is used to store kernel's CS\DS.
111  * GDT[4]\[5] is used to store current running task's CS\DS.
112  * so you can calculate how many tasks will be supported in 80386 in theory.
113  * 4000+!
114  *
115  * here, we encounter the tss struct definition.
116  *
117  * we see, it contains some register fields, it is easy to understand, these
118  * registers can be used to describe process's running state.
119  *
120  */
121 struct tss_struct {
122     long    back_link;  /* 16 high bits zero */
123     long    esp0;
124     long    ss0;        /* 16 high bits zero */
125     long    esp1;
126     long    ss1;        /* 16 high bits zero */
127     long    esp2;
128     long    ss2;        /* 16 high bits zero */
129     long    cr3;        /**
130                          * hit.zhangjie@gmail.com 
131                          * 2015-04-10 06:43:35 PM
132                          *
133                          * cr3 contains the base address points to the Page 
134                          * Category Table 
135                          */
136     long    eip;
137     long    eflags;
138     long    eax,ecx,edx,ebx;
139     long    esp;
140     long    ebp;
141     long    esi;
142     long    edi;
143     long    es;     /* 16 high bits zero */
144     long    cs;     /* 16 high bits zero */
145     long    ss;     /* 16 high bits zero */
146     long    ds;     /* 16 high bits zero */
147     long    fs;     /* 16 high bits zero */
148     long    gs;     /* 16 high bits zero */
149     long    ldt;    /* 16 high bits zero */
150                     /**
151                      * hit.zhangjie@gmail.com
152                      * 2015-04-10 06:44:29 PM 
153                      * 
154                      * why ldt is contained here, too? it is already contained

336 /**
337  * hit.zhangjie@gmail.com 
338  * 2015-04-10 07:55:17 PM
339  * 
340  * every gdt element takes up 8 bytes. now you should understand why <<4 and
341  * <<3 in following code.
342  * _TSS(n), returns the bytes offset of the (n+1)th tss_struct in gdt, while
343  * _LDT(n), returns the bytes offset of the (n+1)th ldt struct in gdt.
344  */
345 #define _TSS(n) ((((unsigned long) n)<<4)+(FIRST_TSS_ENTRY<<3))
346 #define _LDT(n) ((((unsigned long) n)<<4)+(FIRST_LDT_ENTRY<<3))

354 #define lldt(n) __asm__("lldt %%ax"::"a" (_LDT(n)))
